# Yt-Downloader
Youtube Downloader * ( Tracks + Playlists .zip )

# REQUIREMENTS

    sudo apt update
    sudo apt install yt-dlp ffmpeg zip

# DEPENDANCES NODE.JS

    npm install express cors diskusage

# RUN 

     chmod +x server.js
     node server.js
