# Yt-Downloader
Youtube Downloader * ( Tracks + Playlists .zip )

# REQUIREMENTS

    sudo apt update
    sudo apt install yt-dlp ffmpeg zip unzip

# DEPENDANCES NODE.JS

    npm install express cors 

# RUN 

     chmod +x server.js
     node server.js
